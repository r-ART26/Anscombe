library(ggplot2)
library(datasauRus)
# Crear el gr√°fico
ggplot(datasaurus_dozen, aes(x = x, y = y)) +
  
  # 1. A√±adir los puntos
  # Usamos 'color = dataset' para dar un color √∫nico a cada gr√°fico
  # 'show.legend = FALSE' porque los t√≠tulos de las facetas ya nos dan la info
  geom_point(aes(color = dataset), alpha = 0.7, size = 2, show.legend = FALSE) +
  
  # 2. A√±adir la l√≠nea de regresi√≥n lineal (id√©ntica para todos)
  # 'se = FALSE' oculta el intervalo de confianza para un look m√°s limpio
  geom_smooth(method = "lm", se = FALSE, color = "black", linetype = "dashed", size = 0.8) +
  
  # 3. La funci√≥n clave: crear un sub-gr√°fico para cada 'dataset'
  # 'ncol = 4' organiza la cuadr√≠cula en 4 columnas
  facet_wrap(~ dataset, ncol = 4) +
  
  # 4. Est√©tica y T√≠tulos
  theme_minimal(base_size = 12) +
  labs(
    title = "El 'Datasaurus Dozen'",
    subtitle = "Demostraci√≥n de que estad√≠sticas id√©nticas ocultan patrones visuales √∫nicos",
    x = "Eje X",
    y = "Eje Y",
    caption = "Fuente: Paquete 'datasauRus' en R"
  ) +
  
  # 5. Limpieza visual
  theme(
    # Ocultar los n√∫meros de los ejes (opcional, pero limpia la vista)
    axis.text = element_blank(),
    axis.ticks = element_blank(),
    
    # Poner en negrita los t√≠tulos de cada faceta
    strip.text = element_text(face = "bold", size = 10)
  )
